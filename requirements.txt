# Файл зависимостей Python для проекта «Межевой план»
# Установка: pip install -r requirements.txt

# Базовые библиотеки

# PySide6 — официальные привязки Python для Qt6.
# Используется для создания графического интерфейса (главное окно, диалоги, виджеты).
# Версия 6.8.x обеспечивает стабильную работу на всех платформах.
PySide6 >= 6.8.0, < 6.9.0

# SQLAlchemy — объектно-реляционное отображение (ORM) для работы с базой данных.
# Версия 2.0+ поддерживает современные возможности и асинхронность (не используется).
SQLAlchemy >= 2.0.36, < 2.1.0

# lxml — быстрая и полная библиотека для обработки XML и HTML.
# Используется для парсинга XML-схем, генерации XML-документов и валидации по XSD.
lxml >= 5.3.0, < 5.4.0

# Shapely — библиотека для работы с геометрическими объектами.
# Используется для создания полигонов из координат, проверки пересечений,
# вычисления площади и буферных зон.
Shapely >= 2.0.6, < 2.1.0

# pyproj — интерфейс к библиотеке PROJ для преобразования координат между системами.
# Может использоваться для пересчёта координат из одной системы в другую.
pyproj >= 3.6.1, < 3.7.0

# pytest — фреймворк для модульного и интеграционного тестирования.
# Используется для написания и запуска тестов всех уровней.
pytest >= 8.3.0, < 9.0.0

# pytest-qt — плагин для тестирования PySide6-приложений.
# Позволяет эмулировать действия пользователя и проверять состояние GUI.
pytest-qt >= 4.4.0, < 5.0.0

# coverage — инструмент для измерения покрытия кода тестами.
# Рекомендуется для оценки качества тестирования.
coverage >= 7.6.0, < 8.0.0

# (Опционально) flake8 — линтер для проверки стиля кода.
# flake8 >= 7.1.0, < 8.0.0

# (Опционально) black — автоматический форматтер кода.
# black >= 24.8.0, < 25.0.0

# (Опционально) mypy — статический анализатор типов.
# mypy >= 1.11.0, < 2.0.0
